document.addEventListener('DOMContentLoaded', () => {
    // 1. Данные о планетах
    const planetFacts = {
        mercury: {
            name: "Меркурий",
            facts: "Меркурий - самая маленькая планета и ближайшая к Солнцу. Год на Меркурии длится всего 88 земных дней.",
            // Добавьте больше данных (изображение, подробные факты)
        },
        venus: {
            name: "Венера",
            facts: "Венера - самая горячая планета в Солнечной системе из-за парникового эффекта. Она вращается в обратном направлении.",
        },
        earth: {
            name: "Земля",
            facts: "Земля - единственная известная планета, на которой есть жизнь. Около 71% поверхности покрыто водой.",
        },
        // Добавьте данные для Марса, Юпитера, Сатурна, Урана, Нептуна
    };

    // 2. Элементы DOM
    const planets = document.querySelectorAll('.planet');
    const modal = document.getElementById('modal');
    const modalPlanetName = document.getElementById('modal-planet-name');
    const modalPlanetFacts = document.getElementById('modal-planet-facts');
    const closeButton = document.querySelector('.close-button');

    // 3. Обработчик клика по планете
    planets.forEach(planet => {
        planet.addEventListener('click', () => {
            const planetKey = planet.getAttribute('data-planet');
            const data = planetFacts[planetKey];

            if (data) {
                modalPlanetName.textContent = data.name;
                modalPlanetFacts.textContent = data.facts;
                modal.style.display = 'block'; // Показать модальное окно
            }
        });
    });

    // 4. Закрытие модального окна
    closeButton.addEventListener('click', () => {
        modal.style.display = 'none';
    });

    // Закрытие при клике вне модального окна
    window.addEventListener('click', (event) => {
        if (event.target === modal) {
            modal.style.display = 'none';
        }
    });
});
